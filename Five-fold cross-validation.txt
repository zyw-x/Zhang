import numpy as np
import pandas as pd
import tensorflow as tf
from tensorflow.keras.models import Sequential
from tensorflow.keras.layers import Conv1D, Dense, Flatten, MaxPooling1D
from tensorflow.keras.optimizers import Adam
from tensorflow.keras.callbacks import EarlyStopping
from sklearn.model_selection import KFold
from scipy.stats import pearsonr
from tensorflow.keras.initializers import LecunNormal

data_mate = pd.read_csv('/path_to_data/data.csv')
y = data_mate['total'].values.astype(float)
X = data_mate.iloc[:, 2:].values.astype(float)
X = np.expand_dims(X, axis=2)

print(X.shape, y.shape)

def build_model(input_shape):
    model = Sequential()

    model.add(Conv1D(64, 2, activation='selu', input_shape=input_shape, kernel_initializer=LecunNormal()))
    model.add(Conv1D(64, 2, activation='selu', kernel_initializer=LecunNormal()))
    model.add(MaxPooling1D(2))

    model.add(Conv1D(64, 2, activation='selu', kernel_initializer=LecunNormal()))
    model.add(Conv1D(64, 2, activation='selu', kernel_initializer=LecunNormal()))
    model.add(MaxPooling1D(2))

    model.add(Conv1D(64, 2, activation='selu', kernel_initializer=LecunNormal()))
    model.add(Conv1D(64, 2, activation='selu', kernel_initializer=LecunNormal()))
    model.add(MaxPooling1D(2))

    model.add(Flatten())
    model.add(Dense(64, activation='selu', kernel_initializer=LecunNormal()))
    model.add(Dense(1, activation='linear'))

    opt = Adam(learning_rate=1e-4)
    model.compile(optimizer=opt, loss='mse')
    return model



kf = KFold(n_splits=5, shuffle=True, random_state=42)

r_list = []
mse_list = []
mae_list = []
fold = 1

for train_idx, test_idx in kf.split(X):

    print(f"\n===== Fold {fold} =====")

    X_train, X_test = X[train_idx], X[test_idx]
    y_train, y_test = y[train_idx], y[test_idx]

    model = build_model(input_shape=(X_train.shape[1], 1))

    es = EarlyStopping(
        monitor='val_loss',
        patience=10,
        restore_best_weights=True,
        verbose=0
    )

    model.fit(
        X_train, y_train,
        validation_split=0.2,
        epochs=100,
        batch_size=256,
        callbacks=[es],
        verbose=0
    )

    y_pred = model.predict(X_test).flatten()

    r, _ = pearsonr(y_test, y_pred)
    r_list.append(r)

    mse = np.mean((y_test - y_pred) ** 2)
    mse_list.append(mse)
    
    mae = np.mean(np.abs(y_test - y_pred))
    mae_list.append(mae)

    print(f"Fold {fold} Pearson r = {r:.4f}, MSE = {mse:.4f}, MAE = {mae:.4f}")

    fold += 1
